<p>large</p>
<v-select default-value="Orange" large style="width:200px" @change="onSlectChange">

  <v-option value="Cherry">Cherry</v-option>
  <v-option value="Orange">OrangeText</v-option>

new Vue({

  components: {

      single: ''
    }
  },
  methods: {
    onSlectChange (obj) {
      this.single = obj.value;
    }
  }
})
